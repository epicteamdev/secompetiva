def dfs(grafo, inicio):
    visitado, stack = [], [inicio]
    while stack:
        vertex = stack.pop()
        if vertex not in visitado:
            visitado.append(vertex)
            print vertex
            stack.extend(v for v in grafo[vertex] if v not in visitado and v not in stack)
            print stack
            print "visitados: ", visitado
    return visitado
